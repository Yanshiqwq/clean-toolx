# clean-toolx

~~一个屑初中生瞎写的阴间程序()~~

一个普通的命令行清理工具...

插件格式(json):
```
{
	"name": "插件名",
	"author": "插件作者",
	"version_name": "版本名",
	"version_code": 版本号,
	"platform": "操作系统",
	"rules": [
		{
			"name": "规则名",
			"warning_level": 警告值,
			"folders": [
				{
					"path": "文件夹路径",
					"extension": "文件后缀或者'*'"
				},
				......
			],
			"files": [
				{
					"path": "文件/文件夹路径",
					"strict": true
				},{
					"path": "文件/文件夹路径中包含的字符串",
					"strict": false
				},
				......
			],
			"excludes": [
				{
					"path": "排除文件/文件夹路径",
					"strict": true
				},{
					"path": "排除路径中包含字符串的文件/文件夹",
					"strict": false
				},
				......
			]
		},
		......
	]
}
```
